ES6 Modules Named Exports

moduleA.js

// Export moduleA
export const sqrt = Math.sqrt;
export function sqr(x) {
  return x * x;
}
export function diag(x, y) {
  return sqrt(sqr(x) + sqr(y));
}

moduleB.js

// Import individual modules
import { sqr, diag } from './moduleA';
console.log(sqr(11)); // 121
console.log(diag(4, 3)); // 5

moduleC.js

// Import all modules as an object
import { * } as moduleA from './moduleA';
console.log(moduleA.sqr(11)); // 121
console.log(moduleA.diag(4, 3)); // 5
