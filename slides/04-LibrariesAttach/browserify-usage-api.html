API

build.js

var fs = require('fs');
var browserify = require('browserify');

// Create an instance of browserify.
var b = browserify();

// Add an entry file.
b.add('entry.js');

// Bundle entry.js and all its dependencies.
b.bundle()
  // Write the output to disk.
  .pipe(fs.createWriteStream('bundle.js')
);

package.json

"scripts": {
  "build": "node build"
}

Usage

$ npm run build
